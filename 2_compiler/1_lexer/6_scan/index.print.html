类定义

 1struct Lexer
 2{
 3    Lexer(CharStream &ics) : pos(0,1), stream(ics) {}
 4
 5    // 获取词元
 6    Token getNext();
 7    // 退回词元
 8    void backToken(Token &tok)
 9    {
10        stream.back(Char(tok.pos.pos, 0));
11        pos = tok.pos;
12    }
13
14private:
15    // 跳过空白字符
16    inline void SkipWhiteSpaces()
17    {
18        Char ch = stream.next();
19        while (ch.isWhiteSpace())
20        {
21            if (ch.val == '\n')
22                pos.line++;
23            ch = stream.next();
24        }
25        stream.back(ch);
26    }
27    // 解析一个单词
28    Token GetToken();
29
30    // 解析标识符和普通关键字
31    Token GetIdent();
32    // 解析注解关键字
33    Token GetAnnotate();
34    // 解析字符常量
35    Token GetChar();
36    // 解析字符串常量
37    Token GetString(bool escape);
38    // 解析整数或者浮点常量
39    Token GetDigit();
40    // 解析浮点数
41    Token GetFltNumber(Token &tok);
42    // 解析操作符或者分隔符
43    Token GetOperator();
44    // 解析分隔符
45    Token GetSeparator(Token &tok,uint32_t val);
46    // 解析一个单行注释
47    Token GetSinglineComment(Token &tok);
48    // 解析多行注释
49    Token GetMultilineComment(Token &tok);
50
51private:
52    TokPos pos;         // 最后一个token的位置
53    CharStream &stream; // 输入流
54};

解析主体

 1//获取下一个词元
 2Token Lexer::getNext()
 3{
 4    while (true)
 5    {
 6        auto tok = GetToken();
 7        if (tok.kind == TkKind::SingleComments || tok.kind == TkKind::MultipleComments)
 8            continue;
 9        pos = tok.pos;
10        return tok;
11    }
12    return Token();
13}
14//解析获取词元
15Token Lexer::GetToken()
16{
17    SkipWhiteSpaces();
18
19    Char ch = stream.next();
20
21    // 解析结束
22    if (ch.isNull())
23        return Token(TkKind::EOS);
24
25    // 标识符
26    if (ch.isalpha_())
27    {
28        stream.back(ch);
29        return GetIdent();
30    }
31    // 数字常量
32    if (ch.isalnum())
33    {
34        stream.back(ch);
35        return GetDigit();
36    }
37
38    switch (ch.val)
39    {
40        // 标注关键字
41    case '@':
42    {
43        Char ch2 = stream.next();
44        if (ch2.val == '"')
45            return GetString(true);
46        stream.back(ch);
47        return GetAnnotate();
48    }
49
50        // 字符常量
51    case '\'':
52        return GetChar();
53
54    // 字符串常量
55    case '"':
56        return GetString(false);
57    }
58    // 分隔符和操作符
59    stream.back(ch);
60    return GetOperator();
61}

 1Token Lexer::GetIdent()
 2{
 3    Token tok;
 4    tok.pos.pos = stream.getpos();
 5    tok.pos.line = pos.line;
 6    Char ch(0, 0);
 7    while (true)
 8    {
 9        ch = stream.next();
10        if (ch.isalnum_() == false)
11        {
12            stream.back(ch);
13            break;
14        }
15        tok.addchar(ch.val);
16    }
17    tok.checkKeyword();
18    return tok;
19}

解析数字常量

  1Token Lexer::GetDigit()
  2{
  3    Token tok(TkKind::UIntLiteral);
  4    Char ch = stream.next();
  5    tok.pos.pos = ch.pos;
  6    tok.pos.line = pos.line;
  7    tok.addchar(ch.val);
  8
  9    if (ch.val == '0')
 10    {
 11        Char ch2 = stream.next();
 12        switch (ch2.val)
 13        {
 14        // 16进制
 15        case 'X':
 16        case 'x':
 17        {
 18            tok.addchar(ch2.val);
 19            while (true)
 20            {
 21                ch2 = stream.next();
 22                if (ch2.isHex() == false)
 23                {
 24                    stream.back(ch2);
 25                    break;
 26                }
 27                tok.addchar(ch2.val);
 28            }
 29
 30            if (tok.txt.empty())
 31                throw ErrorWhat::fmt("%u:%u: 不完整的 16 进制数：%s\n",
 32                                        tok.pos.line, tok.pos.pos, tok.txt.c_str());
 33            tok.val.i64 = std::strtoull(&tok.txt[2], nullptr, 16);
 34            tok.kind = TkKind::UIntLiteral;
 35            return tok;
 36        }
 37        break;
 38            // 2进制
 39        case 'B':
 40        case 'b':
 41        {
 42            tok.addchar(ch2.val);
 43            while (true)
 44            {
 45                ch2 = stream.next();
 46                if (ch2.isBin() == false)
 47                {
 48                    stream.back(ch2);
 49                    break;
 50                }
 51                tok.addchar(ch2.val);
 52            }
 53            if (tok.txt.empty())
 54                throw ErrorWhat::fmt("%u:%u: 不完整的 2 进制数：%s\n",
 55                                        tok.pos.line, tok.pos.pos, tok.txt.c_str());
 56            tok.val.i64 = std::strtoull(&tok.txt[2], nullptr, 2);
 57            tok.kind = TkKind::UIntLiteral;
 58            return tok;
 59        }
 60        break;
 61        // 浮点数
 62        case '.':
 63        {
 64            tok.addchar(ch2.val);
 65            return GetFltNumber(tok);
 66        }
 67        break;
 68        // 8 进制
 69        default:
 70        {
 71            while (true)
 72            {
 73                if (ch2.isOct() == false)
 74                {
 75                    stream.back(ch2);
 76                    break;
 77                }
 78                tok.addchar(ch2.val);
 79                ch2 = stream.next();
 80            }
 81            tok.val.i64 = std::strtoull(&tok.txt[1], nullptr, 8);
 82            tok.kind = TkKind::UIntLiteral;
 83            return tok;
 84        }
 85        break;
 86        }
 87    }
 88
 89    // 10进制或者浮点数
 90    while (true)
 91    {
 92        ch = stream.next();
 93        if (ch.isdigit() == false)
 94            break;
 95        tok.addchar(ch.val);
 96    }
 97    // 浮点数
 98    if (ch.val == '.')
 99    {
100        tok.addchar(ch.val);
101        return GetFltNumber(tok);
102    }
103    stream.back(ch);
104    // 10进制
105    tok.val.i64 = std::strtoull(tok.txt.c_str(), nullptr, 10);
106    return tok;
107}

109Token Lexer::GetFltNumber(Token &tok)
110{
111    tok.kind = TkKind::Flt64Literal;
112    Char ch = stream.next();
113    while (true)
114    {
115        if (ch.isdigit() == false)
116            break;
117        tok.addchar(ch.val);
118        ch = stream.next();
119    }
120    // 普通浮点数
121    if (ch.val != 'E' && ch.val != 'e')
122    {
123        // 32位浮点
124        if (ch.val == 'F' || ch.val == 'f')
125        {
126            tok.addchar(ch.val);
127            tok.kind = TkKind::Flt32Literal;
128            tok.val.f32 = std::strtof(tok.txt.c_str(), nullptr);
129            return tok;
130        }
131        // 64位浮点
132        stream.back(ch);
133        tok.val.f64 = std::strtod(tok.txt.c_str(), nullptr);
134        return tok;
135    }
136    // 科学计数法
137    tok.addchar(ch.val);
138    ch = stream.next();
139    // 查看符号位
140    if (ch.val == '+' || ch.val == '-')
141        tok.addchar(ch.val);
142    else if (ch.isdigit() == false)
143        throw ErrorWhat::fmt("%u:%u: 不完整的科学计数法：%s\n",
144                                tok.pos.line, tok.pos.pos, tok.txt.c_str());
145    while (true)
146    {
147        ch = stream.next();
148        if (ch.isdigit() == false)
149        {
150            stream.back(ch);
151            break;
152        }
153        tok.addchar(ch.val);
154    }
155    tok.val.f64 = std::strtod(tok.txt.c_str(), nullptr);
156    return tok;
157}

解析字符串常量

 1Token Lexer::GetString(bool escape)
 2{
 3    Token tok(escape == true ? TkKind::StringLiteral2 : TkKind::StringLiteral);
 4    Char ch = stream.next();
 5    tok.pos.pos = ch.pos;
 6    tok.pos.line = pos.line;
 7    if (escape == false)
 8    {
 9        while (true)
10        {
11            if (ch.isNull())
12                throw ErrorWhat::fmt("%u:%u: 文件意外结束！缺少符号 ‘\"’\n",
13                                        tok.pos.line, tok.pos.pos);
14
15            // 转义字符
16            if (ch.val == '\\')
17            {
18                Char ch2 = stream.next();
19                if (ch2.val == '"')
20                {
21                    tok.addchar(ch2.val);
22                    continue;
23                }
24                stream.back(ch2);
25            }
26            // 是否结束
27            else if (ch.val == '"')
28                break;
29            tok.addchar(ch.val);
30            ch = stream.next();
31        }
32    }
33    else
34    {
35        while (true)
36        {
37            if (ch.isNull())
38                throw ErrorWhat::fmt("%u:%u: 文件意外结束！缺少符号 ‘\"’\n",
39                                        tok.pos.line, tok.pos.pos);
40            // 是否结束
41            else if (ch.val == '"')
42                break;
43            tok.addchar(ch.val);
44            ch = stream.next();
45        }
46    }
47
48    return tok;
49}

解析字符常量

 1Token Lexer::GetChar()
 2{
 3    Token tok(TkKind::CharLiteral);
 4    Char ch = stream.next();
 5    tok.pos.pos = ch.pos;
 6    tok.pos.line = pos.line;
 7    tok.addchar(ch.val);
 8    // 转义字符
 9    if (ch.val == '\\')
10    {
11        ch = stream.next();
12        tok.addchar(ch.val);
13        /* // 单字节字符
14            if (ch.val == 'x' || ch.val == 'X')
15            {
16                throw ErrorWhat::fmt("%u:%u: 暂不支持 unicode 转义\n", tok.pos.line, tok.pos.pos);
17            }
18            // unicode转义
19            else if (ch.val == 'u' || ch.val == 'U')
20            {
21                throw ErrorWhat::fmt("%u:%u: 暂不支持 unicode 转义\n", tok.pos.line, tok.pos.pos);
22            }
23            else*/
24        // 普通转义字符
25        {
26            tok.val.i32 = ch.escape();
27            if (-1 == tok.val.i32)
28                throw ErrorWhat::fmt("%u:%u: 未知转义字符 '\\%s'\n", tok.pos.line, tok.pos.pos, tok.txt.c_str());
29        }
30        return tok;
31    }
32    // 普通字符
33    tok.val.i32 = ch.val;
34    ch = stream.next();
35    if (ch.val != '\'')
36        throw ErrorWhat::fmt("%u:%u: 缺少符号 ‘'’\n", tok.pos.line, tok.pos.pos);
37    return tok;
38}

解析操作符

  1Token Lexer::GetOperator()
  2{
  3    Token tok(TkKind::Operator);
  4    Char ch = stream.next();
  5    tok.pos.pos = ch.pos;
  6    tok.pos.line = pos.line;
  7    tok.addchar(ch.val);
  8    // 操作符
  9    switch (ch.val)
 10    {
 11        // 匹配 . 、 ...
 12    case '.':
 13    {
 14        if (stream.match('.', 0) == true && stream.match('.', 1) == true)
 15        {
 16            tok.kind = TkKind::Separator;
 17            ch = stream.next();
 18            tok.addchar(ch.val);
 19            ch = stream.next();
 20            tok.addchar(ch.val);
 21            tok.code.sp = TkSpEnum::Ellipsis;
 22            return tok;
 23        }
 24        tok.code.op = TkOpEnum::Dot;
 25        return tok;
 26    }
 27    break;
 28    // 匹配 + 、++、+=
 29    case '+':
 30    {
 31        if (stream.match('+', 0) == true)
 32        {
 33            ch = stream.next();
 34            tok.addchar(ch.val);
 35            tok.code.op = TkOpEnum::Plus2;
 36        }
 37        else if (stream.match('=', 0) == true)
 38        {
 39            ch = stream.next();
 40            tok.addchar(ch.val);
 41            tok.code.op = TkOpEnum::PlusAssign;
 42        }
 43        else
 44            tok.code.op = TkOpEnum::Plus;
 45        return tok;
 46    }
 47    break;
 48    // 匹配 - 、--、-=、->>、->>=
 49    case '-':
 50    {
 51        if (stream.match('-', 0) == true)
 52        {
 53            ch = stream.next();
 54            tok.addchar(ch.val);
 55            tok.code.op = TkOpEnum::Minus2;
 56        }
 57        else if (stream.match('=', 0) == true)
 58        {
 59            ch = stream.next();
 60            tok.addchar(ch.val);
 61            tok.code.op = TkOpEnum::MinusAssign;
 62        }
 63        else if (stream.match('>', 0) == true && stream.match('>', 1) == true)
 64        {
 65            ch = stream.next();
 66            tok.addchar(ch.val);
 67            ch = stream.next();
 68            tok.addchar(ch.val);
 69            if (stream.match('=', 0) == true)
 70            {
 71                ch = stream.next();
 72                tok.addchar(ch.val);
 73                tok.code.op = TkOpEnum::ShrAAssign;
 74            }
 75            else
 76                tok.code.op = TkOpEnum::ShrA;
 77        }
 78        else
 79            tok.code.op = TkOpEnum::Minus;
 80        return tok;
 81    }
 82    break;
 83    // 匹配 *、*=
 84    case '*':
 85    {
 86        if (stream.match('=', 0) == true)
 87        {
 88            ch = stream.next();
 89            tok.addchar(ch.val);
 90            tok.code.op = TkOpEnum::MultiplyAssign;
 91        }
 92        else
 93            tok.code.op = TkOpEnum::Multiply;
 94        return tok;
 95    }
 96    break;
 97    // 匹配 /、/=、单行注释、多行注释
 98    case '/':
 99    {
100        // 单行注释
101        if (stream.match('/', 0) == true)
102        {
103            ch = stream.next();
104            tok.addchar(ch.val);
105            return GetSinglineComment(tok);
106        } // 多行注释
107        else if (stream.match('*', 0) == true)
108        {
109            ch = stream.next();
110            tok.addchar(ch.val);
111            return GetMultilineComment(tok);
112        }
113        else if (stream.match('=', 0) == true)
114        {
115            ch = stream.next();
116            tok.addchar(ch.val);
117            tok.code.op = TkOpEnum::DivideAssign;
118        }
119        else
120            tok.code.op = TkOpEnum::Divide;
121        return tok;
122    }
123    break;
124    // 匹配 %、%=
125    case '%':
126    {
127        if (stream.match('=', 0) == true)
128        {
129            ch = stream.next();
130            tok.addchar(ch.val);
131            tok.code.op = TkOpEnum::ModulusAssign;
132        }
133        else
134            tok.code.op = TkOpEnum::Modulus;
135        return tok;
136    }
137    break;
138    // 匹配 &、&=、&&
139    case '&':
140    {
141        if (stream.match('=', 0) == true)
142        {
143            ch = stream.next();
144            tok.addchar(ch.val);
145            tok.code.op = TkOpEnum::BitAndAssign;
146        }
147        else if (stream.match('&', 0) == true)
148        {
149            ch = stream.next();
150            tok.addchar(ch.val);
151            tok.code.op = TkOpEnum::LogicalAnd;
152        }
153        else
154            tok.code.op = TkOpEnum::BitAnd;
155        return tok;
156    }
157    break;
158    // 匹配 |、|=、||
159    case '|':
160    {
161        if (stream.match('=', 0) == true)
162        {
163            ch = stream.next();
164            tok.addchar(ch.val);
165            tok.code.op = TkOpEnum::BitOrAssign;
166        }
167        else if (stream.match('|', 0) == true)
168        {
169            ch = stream.next();
170            tok.addchar(ch.val);
171            tok.code.op = TkOpEnum::LogicalOr;
172        }
173        else
174            tok.code.op = TkOpEnum::BitOr;
175        return tok;
176    }
177    break;
178    // 匹配 ~、~=
179    case '~':
180    {
181        if (stream.match('=', 0) == true)
182        {
183            ch = stream.next();
184            tok.addchar(ch.val);
185            tok.code.op = TkOpEnum::BitNotAssign;
186        }
187        else
188            tok.code.op = TkOpEnum::BitNot;
189        return tok;
190    }
191    break;
192    // 匹配 ^、^=
193    case '^':
194    {
195        if (stream.match('=', 0) == true)
196        {
197            ch = stream.next();
198            tok.addchar(ch.val);
199            tok.code.op = TkOpEnum::BitXOrAssign;
200        }
201        else
202            tok.code.op = TkOpEnum::BitXOr;
203        return tok;
204    }
205    break;
206
207    // 匹配 <、<=、<<、<<=、<<)、<<)=
208    case '<':
209    {
210        if (stream.match('=', 0) == true)
211        {
212            ch = stream.next();
213            tok.addchar(ch.val);
214            tok.code.op = TkOpEnum::LE;
215        }
216        else if (stream.match('<', 0) == true)
217        {
218            ch = stream.next();
219            tok.addchar(ch.val);
220            if (stream.match('=', 0) == true)
221            {
222                ch = stream.next();
223                tok.addchar(ch.val);
224                tok.code.op = TkOpEnum::ShlAssign;
225            }
226            else if (stream.match(')', 0) == true)
227            {
228                ch = stream.next();
229                tok.addchar(ch.val);
230                if (stream.match('=', 0) == true)
231                {
232                    ch = stream.next();
233                    tok.addchar(ch.val);
234                    tok.code.op = TkOpEnum::ShlRAssign;
235                }
236                else
237                    tok.code.op = TkOpEnum::ShlR;
238            }
239            else
240                tok.code.op = TkOpEnum::Shl;
241        }
242        else
243            tok.code.op = TkOpEnum::LT;
244        return tok;
245    }
246    break;
247
248    // 匹配 >、>=、>>、>>=、>>(、>>(=
249    case '>':
250    {
251        if (stream.match('=', 0) == true)
252        {
253            ch = stream.next();
254            tok.addchar(ch.val);
255            tok.code.op = TkOpEnum::GE;
256        }
257        else if (stream.match('>', 0) == true)
258        {
259            ch = stream.next();
260            tok.addchar(ch.val);
261            if (stream.match('=', 0) == true)
262            {
263                ch = stream.next();
264                tok.addchar(ch.val);
265                tok.code.op = TkOpEnum::ShrLAssign;
266            }
267            else if (stream.match('(', 0) == true)
268            {
269                ch = stream.next();
270                tok.addchar(ch.val);
271                if (stream.match('=', 0) == true)
272                {
273                    ch = stream.next();
274                    tok.addchar(ch.val);
275                    tok.code.op = TkOpEnum::ShrRAssign;
276                }
277                else
278                    tok.code.op = TkOpEnum::ShrR;
279            }
280            else
281                tok.code.op = TkOpEnum::ShrL;
282        }
283        else
284            tok.code.op = TkOpEnum::GT;
285        return tok;
286    }
287    break;
288
289    // 匹配 !、!=
290    case '!':
291    {
292        if (stream.match('=', 0) == true)
293        {
294            ch = stream.next();
295            tok.addchar(ch.val);
296            tok.code.op = TkOpEnum::NE;
297        }
298        else
299            tok.code.op = TkOpEnum::LogicalNot;
300        return tok;
301    }
302    break;
303    // 匹配 =、==
304    case '=':
305    {
306        if (stream.match('=', 0) == true)
307        {
308            ch = stream.next();
309            tok.addchar(ch.val);
310            tok.code.op = TkOpEnum::EQ;
311        }
312        else
313            tok.code.op = TkOpEnum::Assign;
314        return tok;
315    }
316    break;
317
318    } // 当作分隔符处理
319    return GetSeparator(tok, ch.val);
320}

解析分隔符

 1Token Lexer::GetSeparator(Token &tok, uint32_t val)
 2{
 3    tok.kind = TkKind::Separator;
 4    switch (val)
 5    {
 6        // 匹配 [
 7    case '[':
 8    {
 9        tok.code.sp = TkSpEnum::OpenBracket;
10        return tok;
11    }
12    break;
13    // 匹配 ]
14    case ']':
15    {
16        tok.code.sp = TkSpEnum::CloseBracket;
17        return tok;
18    }
19    break;
20    // 匹配 (
21    case '(':
22    {
23        tok.code.sp = TkSpEnum::OpenParen;
24        return tok;
25    }
26    break;
27    // 匹配 )
28    case ')':
29    {
30        tok.code.sp = TkSpEnum::CloseParen;
31        return tok;
32    }
33    break;
34    // 匹配 {
35    case '{':
36    {
37        tok.code.sp = TkSpEnum::OpenBrace;
38        return tok;
39    }
40    break;
41    // 匹配 }、};
42    case '}':
43    {
44        if (stream.match(';', 0) == true)
45        {
46            stream.next();
47            tok.addchar(';');
48            tok.code.sp = TkSpEnum::CBSemiColon;
49        }
50        else
51            tok.code.sp = TkSpEnum::CloseBrace;
52        return tok;
53    }
54    break;
55        // 匹配 :
56    case ':':
57    {
58        tok.code.sp = TkSpEnum::Colon;
59        return tok;
60    }
61    break;
62    // 匹配 ;
63    case ';':
64    {
65        tok.code.sp = TkSpEnum::SemiColon;
66        return tok;
67    }
68    break;
69    // 匹配 ,
70    case ',':
71    {
72        tok.code.sp = TkSpEnum::Comma;
73        return tok;
74    }
75    break;
76    default:
77        throw ErrorWhat::fmt("%u:%u: 未匹配词法规则的字符：%s\n",
78                                tok.pos.line, tok.pos.pos, tok.txt.c_str());
79        break;
80    }
81    return tok;
82}

 1Token Lexer::GetMultilineComment(Token &tok)

 3    tok.kind = TkKind::MultipleComments;
 4    Char ch = stream.next();

 6    {
 7        if (ch.isNull())

10        else if (ch.isChar('\n'))
11        {
12            pos.line++;
13            break;
14        }
15        tok.addchar(ch.val);
16        // 结束
17        if (ch.isChar('*') == true && stream.match('/', 0) == true)
18        {
19            ch = stream.next();
20            tok.addchar(ch.val);
21            break;
22        }
23        ch = stream.next();
24    }
