2.1.5 关键字映射表

类定义

 1struct MapTable
 2{
 3    // 初始化信息
 4    static void init();
 5    // 查找关键字
 6    static TkKeyWord findKeyword(const std::string &szTok)
 7    {
 8        auto find = mapKey.find(szTok);
 9        if (find == mapKey.end())
10            return TkKeyWord::Unknown;
11        return find->second;
12    }
13    // 查找标注关键字
14    static TkAnnotate findAnnword(const std::string &szTok)
15    {
16        auto find = mapAnn.find(szTok);
17        if (find == mapAnn.end())
18            return TkAnnotate::Unknown;
19        return find->second;
20    }
21    // 查找操作符优先级
22    static int32_t findOpPriority(TkOpEnum op)
23    {
24        auto find = mapOp.find(op);
25        if (find == mapOp.end())
26            return INT32_MAX;
27        return find->second;
28    }
29
30private:
31    static std::map<std::string, TkKeyWord> mapKey;  // 普通关键字映射表
32    static std::map<std::string, TkAnnotate> mapAnn; // 注解关键字映射表
33    static std::map<TkOpEnum, int32_t> mapOp;        // 操作符优先级映射
34
35    static void initMapKey(); // 初始化普通关键字映射表
36    static void initMapAnn(); // 初始化注解关键字映射表
37    static void initMapOp();  // 初始化操作符优先级映射
38};
