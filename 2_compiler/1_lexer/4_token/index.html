2.1.4 词元(Token)定义

词元类别

 1// 词元类别
 2enum class TkKind : uint16_t
 3{
 4    Unknown,          // 未知
 5    SingleComments,   // 单行注释
 6    MultipleComments, // 多行注释
 7    Annotate,         // 注解
 8    KeyWord,          // 关键字
 9    Identifier,       // 标志符
10    StringLiteral,    // 存在转义的字符串常量
11    StringLiteral2,   // 不转义的字符串常量
12    CharLiteral,      // 字符常量
13    SIntLiteral,      // 有符号整形字符串常量
14    UIntLiteral,      // 无符号整形字符串常量
15    Flt32Literal,     // 浮点字符串常量
16    Flt64Literal,     // 浮点字符串常量
17    Separator,        // 分隔符
18    Operator,         // 操作符
19    EOS,              // 字符流结束
20};

分隔符

 1// 分隔符
 2enum class TkSpEnum : uint16_t
 3{
 4    Unknown,      // 未知
 5    OpenBracket,  // [ 左方括号
 6    CloseBracket, // ] 右方括号
 7    OpenParen,    // ( 左括号
 8    CloseParen,   // ) 右括号
 9    OpenBrace,    // { 左花括号
10    CloseBrace,   // } 右花括号
11    CBSemiColon,  // }; 右花括号和分号
12    Colon,        // : 冒号
13    SemiColon,    // ; 分号
14    Comma,        // , 逗号
15    Ellipsis,     // ... 省略号
16};

 1// 操作符
 2enum class TkOpEnum : uint16_t
 3{
 4    Unknown, // 未知

 6    Dot, //. 成员访问符
 7
 8    Plus2,  // ++  自增

11    Plus,     // + 加法

13    Multiply, // * 乘法
14    Divide,   // / 除法
15    Modulus,  // % 求余
16
17    BitAnd, // & 位与
18    BitOr,  // | 位或
19    BitNot, // ~ 位非
20    BitXOr, // ^ 位异或
21
22    Shl,  // <<   左移
23    ShrL, // >>   逻辑右移
24    ShrA, // ->>  算术右移
25    ShlR, // <<)  循环左移
26    ShrR, // >>(  循环右移
27
28    LogicalNot, // !  逻辑非
29    LogicalAnd, // && 逻辑与
30    LogicalOr,  // || 逻辑或
31
32    Assign, // =   赋值
33
34    PlusAssign,     // += 复合加法
35    MinusAssign,    // -= 复合减法
36    MultiplyAssign, // *= 复合乘法
37    DivideAssign,   // /= 复合除法
38    ModulusAssign,  // %= 复合求余
39
40    BitAndAssign, // &= 复合位与
41    BitOrAssign,  // |= 复合位或
42    BitNotAssign, // ~= 复合位非
43    BitXOrAssign, // ^= 复合位异或
44
45    ShlAssign,  // <<=   复合左移
46    ShrLAssign, // >>=   复合逻辑右移
47    ShrAAssign, // ->>=  复合算术右移
48    ShlRAssign, // <<)=  复合循环左移
49    ShrRAssign, // (>>=  复合循环右移
50
51    EQ, // == 等于
52    NE, // != 不等于
53    GT, // >  小于
54    GE, // >= 小于等于
55    LT, // <  大于
56    LE, // <= 大于等于
57};
