测试程序

 1#include <iostream>
 2
 3#include "lexer/Lexer.hpp"
 4#include "mapTable.hpp"
 5#include "utils/errorWhat.hpp"
 6#include "utils/print.hpp"
 7using namespace std;
 8
 9int main(int argc, char **argv)
10{
11    system("chcp 65001");
12    system("cls");
13
14    air::MapTable::init();
15    air::CharStream stream;
16    std::string file = "E:\\airlang\\Dev\\dev\\airlang\\lexer.test.air";
17    if (stream.open(file) == false)
18    {
19        air::Error("%s: 读取文件失败！\n", file.c_str());
20        return 0;
21    }
22    air::Lexer lexer(stream);
23    try
24    {
25        while (true)
26        {
27            auto tok = lexer.getNext();
28            if (tok.kind == air::TkKind::EOS)
29                break;
30            air::Print("%u:%u\t%u:%u\t %s\n",
31                       tok.pos.line, tok.pos.pos,
32                       (uint16_t)tok.kind, (uint16_t)tok.code.key,
33                       tok.txt.c_str());
34            if (tok.kind == air::TkKind::UIntLiteral)
35                air::Print("\t%llu\n", tok.val.i64);
36            else if (tok.kind == air::TkKind::Flt32Literal)
37                air::Print("\t%f\n", tok.val.f32);
38            else if (tok.kind == air::TkKind::Flt64Literal)
39                air::Print("\t%lf\n", tok.val.f64);
40        }
41    }
42    catch (air::ErrorWhat what)
43    {
44        air::Error("错误：%s\n", what.what.c_str());
45    }
46
47    return 0;
48}
